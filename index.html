let userMarker = null; navigator.geolocation.watchPosition(pos => { const lat = pos.coords.latitude; const lng = pos.coords.longitude; debug(`現在地: ${lat}, ${lng}`); if (!userMarker) { userMarker = L.circleMarker([lat, lng], { radius: 10, color: "#136aec", fillColor: "#2a93ff", fillOpacity: 0.7 }).addTo(map); } else { userMarker.setLatLng([lat, lng]); } map.setView([lat, lng]); });
